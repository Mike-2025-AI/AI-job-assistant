## 3. 职位列表页实现（jobs.html）

### 3.1 数据加载

职位数据内联在 HTML 中，以 JS 全局变量 `const JOBS = [...]` 的形式嵌入。

生成流程：
```python
# 1. 读取 CSV
with open('liepin_job_list.csv', 'rb') as f:
    raw = f.read().decode('utf-8-sig')  # 去掉 BOM

# 2. 解析为 JSON 数组
jobs = json.dumps(list(csv.DictReader(raw.splitlines())), ensure_ascii=False)

# 3. 注入 HTML 模板
html = html_template.replace('const JOBS = []', f'const JOBS = {jobs}')
```

### 3.2 URL 参数编码

由于职位数据需传递给 `analyze.html`，使用 base64 编码：

```javascript
function encodeJob(job) {
  const json = JSON.stringify(job);
  const bytes = new TextEncoder().encode(json);
  let binary = '';
  for (let i = 0; i < bytes.length; i++) {
    binary += String.fromCharCode(bytes[i]);
  }
  return btoa(binary);
}
```

> 注意：避免使用 spread operator `[...bytes]`，在某些浏览器版本中会有问题。

### 3.3 跳转方式


### 4.2 简历文件解析

**PDF 解析（pdf.js）：**
```javascript
async function parsePDF(arrayBuffer) {
  const pdf = await pdfjsLib.getDocument({ data: arrayBuffer }).promise;
  let text = '';
  for (let i = 1; i <= pdf.numPages; i++) {
    const page = await pdf.getPage(i);
    const content = await page.getTextContent();
    text += content.items.map(item => item.str).join(' ') + '\n';
  }
  return text;
}
```

**DOCX 解析（Mammoth.js）：**
```javascript
async function parseDOCX(arrayBuffer) {
  const result = await mammoth.extractRawText({ arrayBuffer });
  return result.value; // 纯文本
}
```

### 4.3 Coze API 调用流程

**Step 1：发起聊天**
```javascript
const initResp = await fetch(`${COZE_API_BASE}/v3/chat`, {
  method: 'POST',
  headers: {
    'Authorization': `Bearer ${COZE_API_TOKEN}`,
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    bot_id: BOT_ID,
    user_id: 'job-user-' + Date.now(),
    stream: false,
    auto_save_history: false,
    additional_messages: [{
      role: 'user',
      content: `简历内容：${resumeText}\n\n岗位信息：${jobText}`,
      content_type: 'text'
    }]
  })
});
const { data: { chat_id, conversation_id } } = await initResp.json();
```

**Step 2：轮询结果**
```javascript
async function pollResult(chat_id, conversation_id) {
  const maxAttempts = 30;  // 60s 超时
  for (let i = 0; i < maxAttempts; i++) {
    await sleep(2000);
    const retResp = await fetch(
      `${COZE_API_BASE}/v3/chat/retrieve?chat_id=${chat_id}&conversation_id=${conversation_id}`,
      { headers: { 'Authorization': `Bearer ${COZE_API_TOKEN}` } }
    );
    const { data: { status } } = await retResp.json();
    if (status === 'completed') {
      return await fetchMessages(conversation_id, chat_id);
    }
  }
  throw new Error('分析超时，请重试');
}
```

**Step 3：解析消息**
```javascript
async function fetchMessages(conversation_id, chat_id) {
  const msgResp = await fetch(
    `${COZE_API_BASE}/v3/chat/message/list?chat_id=${chat_id}&conversation_id=${conversation_id}`,
    { headers: { 'Authorization': `Bearer ${COZE_API_TOKEN}` } }
  );
  const { data: { items } } = await msgResp.json();
  const final = items.find(m => m.role === 'assistant' && m.type === 'answer');
  return final.content; // AI 回复文本
}
```

### 4.4 报告渲染

AI 返回的文本格式为 Markdown，支持手动解析各模块：

```javascript
function parseReport(text) {
  // 综合评分
  const scoreMatch = text.match(/综合匹配[分度][：:]\s*(\d+)/);
  const score = scoreMatch ? parseInt(scoreMatch[1]) : 0;

  // 六维评分（正则匹配）
  const dims = ['技能匹配', '经验匹配', '学历匹配', '行业匹配', '薪资匹配', '整体竞争力'];
  dims.forEach(dim => {
    const m = text.match(new RegExp(`${dim}[：:]\s*(\\d+)`));
    if (m) results[dim] = parseInt(m[1]);
  });

  // 亮点 / 弱项
  const strengths = extractBetween(text, '匹配亮点', '差距分析');
  const gaps = extractBetween(text, '差距分析', '改进建议');

  return { score, results, strengths, gaps };
}
```
